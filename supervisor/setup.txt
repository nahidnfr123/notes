sudo apt update && sudo apt install supervisor

sudo mkdir -p /var/www/log
sudo chown -R $USER_NAME:$USER_NAME /var/www/log
sudo chmod -R 777 /var/www/log

sudo nano /etc/supervisor/conf.d/sc-queue.conf
sudo nano /etc/supervisor/conf.d/sc-scheduler.conf



sudo supervisorctl reread
sudo supervisorctl update

sudo supervisorctl start sc-queue:*
sudo supervisorctl start sc-scheduler


or
sudo supervisorctl start 'sc-queue:*'



# create alias "supervisor refresh" : this command will update the newly added supervisor config and restrart all.
echo '
supervisor() {
    if [ "$1" = "refresh" ]; then
        sudo supervisorctl reread
        sudo supervisorctl update
        sudo supervisorctl restart all
    else
        command supervisor "$@"
    fi
}
' >> ~/.bashrc && source ~/.bashrc
