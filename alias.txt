# push to github: git add . && git commit -m "commit message" && git pull && git push
# usage: push "commit message"
push() {
  if [ -z "$1" ]; then
    echo "❌ Commit message required!"
    echo "Usage: push \"your commit message\""
    return 1
  fi

  git add .
  git commit -m "$1"
  git pull
  git push
}
